#include <iostream>
using namespace std;

int main()
{
double* d = new double[3];
if(d == NULL)
{
     cout << "Error: Insufficient Memory\n";
     exit(0);
}
for(unsigned int i = 0; i< 3; i++) {
d[i] = 1.5 + i;
}

for(int i = 2; i>= 0; i--) {
cout<< d[i] <<endl;
}
delete d;
d = NULL;

system("pause");
}
